SORTED=$(echo "$RAW" | sort -n)
N=$(echo "$SORTED" | wc -l | tr -d ' ')
RANK=$(awk -v n="$N" 'BEGIN { r = n * 0.95; rr = (r == int(r)) ? r : int(r) + 1; print rr }')
P95=$(echo "$SORTED" | sed -n "${RANK}p")
